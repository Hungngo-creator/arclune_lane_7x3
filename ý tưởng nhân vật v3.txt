# Final stats
Final[S≠SPD] = ( CLASS_BASE[class][S] + TP[S] * Δ_S ) * RANK_MULT[rank]
Final[SPD]   =   CLASS_BASE[class][SPD]              # SPD không nhân RANK_MULT

# 1) Cơ sở mỗi hit (basic/skill)
N = 0.50        # WIL 50%
X = 0.50        # ATK 50%
powMul = k      # hệ số skill/ult (basic thường = 1.0)

base_phys = X * ATK * powMul
base_mag  = N * WIL * powMul

# 2) Giáp/kháng + xuyên (áp RIÊNG từng nửa)
ARM_eff = max(0, ARM * (1 - defPen))
RES_eff = max(0, RES * (1 - defPen))

phys_after = floor( base_phys * (1 - ARM_eff) )
mag_after  = floor( base_mag  * (1 - RES_eff) )

total_after_def = phys_after + mag_after

# 3) “% gây trên TỔNG” của attacker + “% nhận trên TỔNG” của target
dealMul_final = clamp( 1 + ΣdealBonus,                 0.50, 2.00 )   # -50%..+100%
recvMul_final = clamp( (1 - ΣrecvReduce) * (1+Σvuln),  0.25, 1.75 )   # -75%..+75%

raw = floor( total_after_def * dealMul_final * recvMul_final )

# 4) Khiên & hook sau va chạm
final = applyShieldThenHP(raw)            # rồi mới reflect/poison/execute/undying

F = followupCap  # mặc định 2; riêng Kiếm Tốc #20 = 3
hit_count_per_action = 1 + F   # 1 đòn thường + F follow-up (F KHÔNG tính đòn thường)
TotalDamagePerAction ≈ (1+F) * DamagePerHit   # ước tính nhanh

DISPLAY_DIVISOR = 10
shownDamage = round(raw / DISPLAY_DIVISOR)
shownHP     = round(HP_engine / DISPLAY_DIVISOR)

Tính sát thương được cộng như thế nào? Chúng ta chưa có công thức tính sát thương
Giảm % sát thương nhận vào /tổng sát thương nhận vào cũng thế.
Đòn đánh thường của mọi char n% và x% cần căn cứ vào rank, chúng ta cần dựa vào rank budget để gán cho N và X số % cụ thể sau đó tính ra sát thương từ kĩ năng vì kĩ năng tính theo % đòn đánh thường. Chúng ta nên có một tỉ lệ N và X% cố định dựa theo rank.
Kĩ năng không cần hồi chiêu vì đã có max aether nên chúng ta chỉ cần set cost của kĩ năng tốn aether hợp lý là được, vì game theo turn nên không thể spam liên tục được.
